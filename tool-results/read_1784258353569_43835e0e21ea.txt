   327→                  </Button>
   328→                </div>

   347→        onOpenChange={(o) => { setDetailOpen(o); if (!o) setSelectedShipmentId(null); }}
   348→      />
   349→    </div>

   353→// ---------- Create dialog ----------

   359→  onCreated: (s: ShipmentListItem) => void;

   363→    senderId: "", receiverId: "", originAddress: "", originCity: "",
   364→    destinationAddress: "", destinationCity: "", weightKg: "", pieces: "1",

   366→    driverId: "", vehicleId: "", notes: "",
   367→  });
   368→
   369→  const { data: customers } = useQuery({
   370→    queryKey: ["customers", "all"],
   371→    queryFn: () => api.get<{ items: { id: string; name: string; city: string }[] }>("/api/customers?pageSize=100"),
   372→    enabled: open,
   373→  });
   374→  const { data: drivers } = useQuery({
   375→    queryKey: ["drivers", "all"],

   383→  });
   384→
   385→  const create = useMutation({
   386→    mutationFn: () => api.post<ShipmentListItem>("/api/shipments", {

   423→                      "flex flex-col items-center gap-1 rounded-lg border p-2 text-xs transition-all",
   424→                      active ? "border-emerald-500 bg-emerald-50 dark:bg-emerald-950/40 text-emerald-700 dark:text-emerald-300" : "hover:bg-muted"
   425→                    )}
   426→                  >
   427→                    <Icon className="h-4 w-4" />
   428→                    {SERVICE_META[s].label}
   429→                  </button>
   430→                );
   431→              })}

   763→function useDebounce<T>(value: T, delay: number): T {
   764→  const [v, setV] = React.useState(value);
   765→  React.useEffect(() => {
   766→    const t = setTimeout(() => setV(value), delay);
   767→    return () => clearTimeout(t);
   768→  }, [value, delay]);
   769→  return v;
   770→}
