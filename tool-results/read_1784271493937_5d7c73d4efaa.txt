    96→export function ShipmentsView() {
    97→  const qc = useQueryClient();
    98→  const { selectedShipmentId, setSelectedShipmentId, shipmentsFilter, setShipmentsFilter } = useAppStore();
    99→
   100→  const [search, setSearch] = React.useState("");
   101→  const [status, setStatus] = React.useState(shipmentsFilter || "all");
   102→  const [priority, setPriority] = React.useState("all");
   103→  const [service, setService] = React.useState("all");
   104→  const [city, setCity] = React.useState("all");
   105→  const [page, setPage] = React.useState(1);
   106→  const [pageSize] = React.useState(12);
   107→  const [createOpen, setCreateOpen] = React.useState(false);
   108→  const [detailOpen, setDetailOpen] = React.useState(false);
   109→  const [selectedIds, setSelectedIds] = React.useState<Set<string>>(new Set());
   110→
   111→  // React to external "open-new-shipment" event
   112→  React.useEffect(() => {
   113→    const handler = () => setCreateOpen(true);
   114→    window.addEventListener("open-new-shipment", handler);
   115→    return () => window.removeEventListener("open-new-shipment", handler);
   116→  }, []);
   117→
   118→  // Open detail when a shipment is selected
   119→  React.useEffect(() => {
   120→    if (selectedShipmentId) setDetailOpen(true);
   121→    else setDetailOpen(false);
   122→  }, [selectedShipmentId]);
   123→
   124→  // Reset filter state when shipmentsFilter changes externally
   125→  React.useEffect(() => {
   126→    if (shipmentsFilter) {
   127→      setStatus(shipmentsFilter);
   128→      setShipmentsFilter(null);
   129→      setPage(1);
   130→    }
   131→  }, [shipmentsFilter, setShipmentsFilter]);
   132→
   133→  const debouncedSearch = useDebounce(search, 350);
   134→
   135→  const queryKey = React.useMemo(
   136→    () => ["shipments", { debouncedSearch, status, priority, service, city, page, pageSize }],
   137→    [debouncedSearch, status, priority, service, city, page, pageSize]
   138→  );
   139→
   140→  const { data, isLoading } = useQuery({
   141→    queryKey,
   142→    queryFn: () => {
   143→      const params = new URLSearchParams({
   144→        page: String(page),
   145→        pageSize: String(pageSize),
   146→      });
   147→      if (debouncedSearch) params.set("q", debouncedSearch);
   148→      if (status !== "all") params.set("status", status);
   149→      if (priority !== "all") params.set("priority", priority);
   150→      if (service !== "all") params.set("service", service);
   151→      if (city !== "all") params.set("city", city);
   152→      return api.get<{ items: ShipmentListItem[]; total: number; totalPages: number }>(`/api/shipments?${params}`);
   153→    },
   154→  });
   155→
   156→  const resetFilters = () => {
   157→    setSearch(""); setStatus("all"); setPriority("all"); setService("all"); setCity("all"); setPage(1);
   158→  };
   159→
   160→  const hasFilters = status !== "all" || priority !== "all" || service !== "all" || city !== "all" || search !== "";
   161→
   162→  const bulkUpdate = useMutation({
   163→    mutationFn: async (newStatus: string) => {
   164→      const ids = Array.from(selectedIds);
   165→      await Promise.all(ids.map((id) => api.patch(`/api/shipments/${id}`, { status: newStatus })));
   166→    },
   167→    onSuccess: () => {
   168→      toast.success(`${selectedIds.size} shipment${selectedIds.size > 1 ? "s" : ""} updated`);
   169→      setSelectedIds(new Set());
   170→      qc.invalidateQueries({ queryKey: ["shipments"] });
   171→      qc.invalidateQueries({ queryKey: ["dashboard"] });
   172→      qc.invalidateQueries({ queryKey: ["notifications"] });
   173→    },
   174→    onError: (e: Error) => toast.error("Bulk update failed", { description: e.message }),
   175→  });
   176→
   177→  const toggleSelect = (id: string) => {
   178→    setSelectedIds((prev) => {
   179→      const next = new Set(prev);
   180→      if (next.has(id)) next.delete(id);
   181→      else next.add(id);
   182→      return next;
   183→    });
   184→  };
   185→
   186→  const toggleSelectAll = () => {
   187→    if (!data) return;
   188→    const allOnPage = data.items.every((s) => selectedIds.has(s.id));
   189→    if (allOnPage) {
   190→      setSelectedIds((prev) => {
   191→        const next = new Set(prev);
   192→        data.items.forEach((s) => next.delete(s.id));
   193→        return next;
   194→      });
   195→    } else {
   196→      setSelectedIds((prev) => {
   197→        const next = new Set(prev);
   198→        data.items.forEach((s) => next.add(s.id));
   199→        return next;
   200→      });
   201→    }
   202→  };
   203→
   204→  const exportSelected = () => {
   205→    if (!data) return;
   206→    const toExport = data.items.filter((s) => selectedIds.has(s.id));
   207→    const rows = [
   208→      ["Tracking #", "Status", "Priority", "Service", "Origin", "Destination", "Sender", "Receiver", "Weight (kg)", "Pieces", "Cost", "Created"],
   209→      ...toExport.map((s) => [
   210→        s.trackingNumber, s.status, s.priority, s.serviceType,
   211→        s.originCity, s.destinationCity, s.sender.name, s.receiver.name,
   212→        String(s.weightKg), String(s.pieces), String(s.cost), new Date(s.createdAt).toISOString(),
   213→      ]),
   214→    ];
   215→    const csv = rows.map((r) => r.map((c) => `"${String(c).replace(/"/g, '""')}"`).join(",")).join("\n");
   216→    const blob = new Blob([csv], { type: "text/csv;charset=utf-8;" });
   217→    const url = URL.createObjectURL(blob);
   218→    const a = document.createElement("a");
   219→    a.href = url;
   220→    a.download = `shipments-export-${new Date().toISOString().slice(0, 10)}.csv`;
   221→    a.click();
   222→    URL.revokeObjectURL(url);
   223→    toast.success(`Exported ${toExport.length} shipment${toExport.length > 1 ? "s" : ""} to CSV`);
   224→  };
   225→
   226→  return (
   227→    <div className="space-y-4">
   228→      {/* Filter bar */}
   229→      <Card>
   230→        <CardContent className="p-4">
   231→          <div className="flex flex-col gap-3 lg:flex-row lg:items-center">
   232→            <div className="relative flex-1">
   233→              <Search className="absolute left-3 top-1/2 h-4 w-4 -translate-y-1/2 text-muted-foreground" />
   234→              <Input
   235→                placeholder="Search tracking #, sender, route…"
   236→                value={search}
   237→                onChange={(e) => { setSearch(e.target.value); setPage(1); }}
   238→                className="pl-9"
   239→              />
   240→            </div>
   241→            <div className="flex flex-wrap items-center gap-2">
   242→              <Select value={status} onValueChange={(v) => { setStatus(v); setPage(1); }}>
   243→                <SelectTrigger className="w-[150px]"><SelectValue placeholder="Status" /></SelectTrigger>
   244→                <SelectContent>
   245→                  <SelectItem value="all">All statuses</SelectItem>
   246→                  {SHIPMENT_STATUSES.map((s) => (
   247→                    <SelectItem key={s} value={s}>{SHIPMENT_STATUS_META[s].label}</SelectItem>
   248→                  ))}
   249→                </SelectContent>
   250→              </Select>
   251→              <Select value={priority} onValueChange={(v) => { setPriority(v); setPage(1); }}>
   252→                <SelectTrigger className="w-[130px]"><SelectValue placeholder="Priority" /></SelectTrigger>
   253→                <SelectContent>
   254→                  <SelectItem value="all">All priority</SelectItem>
   255→                  {PRIORITIES.map((p) => (
   256→                    <SelectItem key={p} value={p}>{PRIORITY_META[p].label}</SelectItem>
   257→                  ))}
   258→                </SelectContent>
   259→              </Select>
   260→              <Select value={service} onValueChange={(v) => { setService(v); setPage(1); }}>
   261→                <SelectTrigger className="w-[140px]"><SelectValue placeholder="Service" /></SelectTrigger>
   262→                <SelectContent>
   263→                  <SelectItem value="all">All services</SelectItem>
   264→                  {SERVICE_TYPES.map((s) => (
   265→                    <SelectItem key={s} value={s}>{SERVICE_META[s].label}</SelectItem>
   266→                  ))}
   267→                </SelectContent>
   268→              </Select>
   269→              <Select value={city} onValueChange={(v) => { setCity(v); setPage(1); }}>
   270→                <SelectTrigger className="w-[150px]"><SelectValue placeholder="City" /></SelectTrigger>
   271→                <SelectContent>
   272→                  <SelectItem value="all">All cities</SelectItem>
   273→                  {VIETNAM_CITIES.map((c) => (
   274→                    <SelectItem key={c} value={c}>{c}</SelectItem>
   275→                  ))}
   276→                </SelectContent>
   277→              </Select>
   278→              {hasFilters && (
   279→                <Button variant="ghost" size="sm" onClick={resetFilters} className="gap-1 text-xs">
   280→                  <X className="h-3.5 w-3.5" /> Clear
   281→                </Button>
   282→              )}
   283→              <Button size="sm" onClick={() => setCreateOpen(true)} className="gap-1.5">
   284→                <Plus className="h-4 w-4" /> New
   285→              </Button>
   286→            </div>
   287→          </div>
   288→        </CardContent>
   289→      </Card>
   290→
   291→      {/* Bulk action bar */}
   292→      {selectedIds.size > 0 && (
   293→        <div className="sticky top-16 z-20 flex flex-wrap items-center gap-2 rounded-lg border border-emerald-200 bg-emerald-50/95 px-4 py-2.5 shadow-sm backdrop-blur dark:border-emerald-900 dark:bg-emerald-950/80">
   294→          <span className="flex items-center gap-2 text-sm font-medium text-emerald-700 dark:text-emerald-300">
   295→            <Checkbox checked={true} onCheckedChange={() => setSelectedIds(new Set())} className="border-emerald-400" />
   296→            {selectedIds.size} selected
   297→          </span>
   298→          <div className="ml-auto flex flex-wrap items-center gap-1.5">
   299→            <Select value="" onValueChange={(v) => { if (v) bulkUpdate.mutate(v); }}>
   300→              <SelectTrigger className="h-8 w-[160px] text-xs">
   301→                <SelectValue placeholder="Update status…" />
   302→              </SelectTrigger>
   303→              <SelectContent>
   304→                {SHIPMENT_STATUSES.map((s) => (
   305→                  <SelectItem key={s} value={s}>{SHIPMENT_STATUS_META[s].label}</SelectItem>
   306→                ))}
   307→              </SelectContent>
   308→            </Select>
   309→            <Button size="sm" variant="outline" className="h-8 gap-1.5 text-xs" onClick={exportSelected}>
   310→              <Download className="h-3.5 w-3.5" /> Export CSV
   311→            </Button>
   312→            <Button size="sm" variant="ghost" className="h-8 gap-1.5 text-xs" onClick={() => setSelectedIds(new Set())}>
   313→              <X className="h-3.5 w-3.5" /> Clear
   314→            </Button>
   315→          </div>
   316→        </div>
   317→      )}
   318→
   319→      {/* Table */}
   320→      <Card>
   321→        <CardContent className="p-0">
   322→          {isLoading ? (
   323→            <div className="space-y-2 p-4">
   324→              {Array.from({ length: 8 }).map((_, i) => <Skeleton key={i} className="h-14 w-full" />)}
   325→            </div>
   326→          ) : !data?.items?.length ? (
   327→            <div className="flex flex-col items-center justify-center gap-3 p-12 text-center">
   328→              <div className="flex h-12 w-12 items-center justify-center rounded-full bg-muted text-muted-foreground">
   329→                <Package className="h-6 w-6" />
   330→              </div>
   331→              <div>
   332→                <p className="text-sm font-medium">No shipments found</p>
   333→                <p className="text-xs text-muted-foreground">Try adjusting filters or create a new shipment.</p>
   334→              </div>
   335→              <Button size="sm" variant="outline" onClick={resetFilters}>Reset filters</Button>
   336→            </div>
   337→          ) : (
   338→            <>
   339→              <div className="overflow-x-auto custom-scroll">
   340→                <Table>
   341→                  <TableHeader>
   342→                    <TableRow className="bg-muted/40 hover:bg-muted/40">
   343→                      <TableHead className="w-[40px]">
   344→                        <Checkbox
   345→                          aria-label="Select all"
   346→                          checked={data.items.length > 0 && data.items.every((s) => selectedIds.has(s.id))}
   347→                          onCheckedChange={toggleSelectAll}
   348→                        />
   349→                      </TableHead>
   350→                      <TableHead className="min-w-[140px]">Tracking #</TableHead>
   351→                      <TableHead className="min-w-[200px]">Route</TableHead>
   352→                      <TableHead>Status</TableHead>
   353→                      <TableHead className="hidden md:table-cell">Priority</TableHead>
   354→                      <TableHead className="hidden lg:table-cell">Driver</TableHead>
   355→                      <TableHead className="text-right">Cost</TableHead>
   356→                      <TableHead className="hidden sm:table-cell">Created</TableHead>
   357→                    </TableRow>
   358→                  </TableHeader>
   359→                  <TableBody>
   360→                    {data.items.map((s) => {
   361→                      const SIcon = SERVICE_ICON[s.serviceType] || Package;
   362→                      const isSelected = selectedIds.has(s.id);
   363→                      return (
   364→                        <TableRow
   365→                          key={s.id}
   366→                          onClick={() => setSelectedShipmentId(s.id)}
   367→                          className={cn("cursor-pointer transition-colors hover:bg-muted/50", isSelected && "bg-emerald-50/60 dark:bg-emerald-950/30")}
   368→                        >
   369→                          <TableCell onClick={(e) => e.stopPropagation()}>
   370→                            <Checkbox
   371→                              aria-label={`Select ${s.trackingNumber}`}
   372→                              checked={isSelected}
   373→                              onCheckedChange={() => toggleSelect(s.id)}
   374→                            />
   375→                          </TableCell>
   376→                          <TableCell>
   377→                            <div className="flex items-center gap-2">
   378→                              <div className="flex h-8 w-8 items-center justify-center rounded-lg bg-emerald-500/10 text-emerald-600 dark:text-emerald-400">
   379→                                <SIcon className="h-4 w-4" />
   380→                              </div>
   381→                              <div className="flex flex-col">
   382→                                <span className="font-mono text-xs font-semibold">{s.trackingNumber}</span>
   383→                                <span className="text-[10px] text-muted-foreground">{SERVICE_META[s.serviceType as keyof typeof SERVICE_META]?.label}</span>
   384→                              </div>
   385→                            </div>
   386→                          </TableCell>
   387→                          <TableCell>
   388→                            <div className="flex items-center gap-1.5 text-sm">
   389→                              <span className="font-medium">{s.originCity}</span>
   390→                              <ArrowRight className="h-3 w-3 text-muted-foreground" />
   391→                              <span className="font-medium">{s.destinationCity}</span>
   392→                            </div>
   393→                            <p className="text-[10px] text-muted-foreground">{s.sender.name} → {s.receiver.name}</p>
   394→                          </TableCell>
   395→                          <TableCell><StatusBadge status={s.status} kind="shipment" /></TableCell>
   396→                          <TableCell className="hidden md:table-cell"><PriorityBadge priority={s.priority} /></TableCell>
   397→                          <TableCell className="hidden lg:table-cell">
   398→                            {s.driver ? (
   399→                              <div className="flex items-center gap-2">
   400→                                <Avatar className="h-7 w-7">
   401→                                  <AvatarFallback className={cn("text-[10px] font-semibold text-white", avatarColorClass(s.driver.avatarColor))}>
   402→                                    {initials(s.driver.name)}
   403→                                  </AvatarFallback>
   404→                                </Avatar>
   405→                                <span className="text-xs">{s.driver.name}</span>
   406→                              </div>
   407→                            ) : (
   408→                              <span className="text-xs text-muted-foreground">Unassigned</span>
   409→                            )}
   410→                          </TableCell>
   411→                          <TableCell className="text-right font-semibold tabular-nums">{formatCurrency(s.cost)}</TableCell>
   412→                          <TableCell className="hidden sm:table-cell text-xs text-muted-foreground">{formatRelativeTime(s.createdAt)}</TableCell>
   413→                        </TableRow>
   414→                      );
   415→                    })}
   416→                  </TableBody>
   417→                </Table>
   418→              </div>
   419→
   420→              {/* Pagination */}
   421→              <div className="flex items-center justify-between gap-3 border-t px-4 py-3">
   422→                <p className="text-xs text-muted-foreground">
   423→                  Showing <span className="font-medium text-foreground">{(page - 1) * pageSize + 1}</span>–
   424→                  <span className="font-medium text-foreground">{Math.min(page * pageSize, data.total)}</span> of{" "}
   425→                  <span className="font-medium text-foreground">{data.total}</span>
   426→                </p>
   427→                <div className="flex items-center gap-1">
   428→                  <Button variant="outline" size="sm" disabled={page <= 1} onClick={() => setPage((p) => Math.max(1, p - 1))} className="gap-1">
   429→                    <ChevronLeft className="h-3.5 w-3.5" /> Prev
   430→                  </Button>
   431→                  <span className="px-2 text-xs text-muted-foreground">{page} / {data.totalPages || 1}</span>
   432→                  <Button variant="outline" size="sm" disabled={page >= data.totalPages} onClick={() => setPage((p) => p + 1)} className="gap-1">
   433→                    Next <ChevronRight className="h-3.5 w-3.5" />
   434→                  </Button>
   435→                </div>
   436→              </div>
   437→            </>
   438→          )}
   439→        </CardContent>
   440→      </Card>
   441→
   442→      {/* Create dialog */}
   443→      <CreateShipmentDialog open={createOpen} onOpenChange={setCreateOpen} onCreated={(s) => {
   444→        setSelectedShipmentId(s.id);
   445→        qc.invalidateQueries({ queryKey: ["shipments"] });
   446→        qc.invalidateQueries({ queryKey: ["dashboard"] });
   447→        toast.success("Shipment created", { description: s.trackingNumber });
   448→      }} />
   449→
   450→      {/* Detail drawer */}
   451→      <ShipmentDetailDrawer
   452→        shipmentId={selectedShipmentId}
   453→        open={detailOpen}
   454→        onOpenChange={(o) => { setDetailOpen(o); if (!o) setSelectedShipmentId(null); }}
   455→      />
   456→    </div>
   457→  );
   458→}

   460→// ---------- Create dialog ----------
   461→function CreateShipmentDialog({
   462→  open, onOpenChange, onCreated,
   463→}: {
   464→  open: boolean;
   465→  onOpenChange: (o: boolean) => void;
   466→  onCreated: (s: ShipmentListItem) => void;
   467→}) {
   468→  const qc = useQueryClient();
   469→  const [form, setForm] = React.useState({
   470→    senderId: "", receiverId: "", originAddress: "", originCity: "",
   471→    destinationAddress: "", destinationCity: "", weightKg: "", pieces: "1",
   472→    description: "", priority: "standard", serviceType: "standard",
   473→    driverId: "", vehicleId: "", notes: "",
   474→  });
   475→
   476→  const { data: customers } = useQuery({
   477→    queryKey: ["customers", "all"],

  1109→function useDebounce<T>(value: T, delay: number): T {
  1110→  const [v, setV] = React.useState(value);
  1111→  React.useEffect(() => {
  1112→    const t = setTimeout(() => setV(value), delay);
  1113→    return () => clearTimeout(t);
  1114→  }, [value, delay]);
  1115→  return v;
  1116→}
